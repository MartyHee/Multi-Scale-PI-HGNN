#   # 1) Create + activate conda environment
#   conda create -n pi_hgnn python=3.10 -y
#   conda activate pi_hgnn
#
#   # 2) Install PyTorch 2.6.0 + CUDA 11.8
#   pip install torch==2.6.0 --index-url https://download.pytorch.org/whl/cu118
#
#   # 3) Install torch_scatter matching torch 2.6 + cu118
#   pip install torch-scatter --index-url https://data.pyg.org/whl/torch-2.6.0+cu118
#
#   # 4) Install torch_geometric and the rest
#   pip install torch_geometric==2.7.0
#   pip install -r requirements.txt
#
# Verify installation:
#   python -c "import torch; print(torch.__version__, torch.cuda.is_available())"
#   python -c "import torch_scatter; print(torch_scatter.__version__)"
#
